## 3. 核心功能模块

### 3.1 博客系统

**位置：** `src/content/blog/`, `src/pages/blog/`

#### 功能特性

- **内容格式**
  - 支持 Markdown 和 MDX 格式
  - 文章元数据管理（标题、描述、日期、标签、封面图）
  - 草稿功能（draft 字段）

- **展示功能**
  - 文章分页（默认每页 8 篇）
  - 标签系统和标签页面
  - 归档页面（按年份分组）
  - 阅读时间自动计算

- **内容增强**
  - 数学公式支持（行内 `$...$` 和块级 `$$...$$`）
  - 代码高亮（语法高亮、行号、差异显示、复制按钮、折叠）
  - 图片灯箱（MediumZoom）

- **互动功能**
  - Waline 评论系统
  - 相关文章推荐
  - 社交分享（微博、X/Twitter、Bluesky）

- **版权保护**
  - 文章版权声明
  - CC BY-NC-SA 4.0 许可

- 包含文章封面图

- 图片路径自动处理
- Pretty Feed 样式表

#### 实现示例

```typescript
export async function GET(context) {
  const blog = await getBlogCollection('blog')
  const sortedPosts = sortMDByDate(blog)

  return rss({
    title: config.theme.title,
    description: config.theme.description,
    site: context.site,
    items: sortedPosts.map((post) => ({
      title: post.data.title,
      pubDate: post.data.publishDate,
      description: post.data.description,
      link: `/blog/${post.id}/`,
    })),
  })
}
```

### 4.3 src/content.config.ts

**内容集合 Schema：**

```typescript
import { defineCollection, z } from 'astro:content'
import { glob } from 'astro/loaders'

const blog = defineCollection({
  loader: glob({
    base: './src/content/blog',
    pattern: '**/*.{md,mdx}'
  }),
  schema: ({ image }) => z.object({
    title: z.string().max(60),
    description: z.string().max(160),
    publishDate: z.coerce.date(),
    updatedDate: z.coerce.date().optional(),
    heroImage: z.object({
      src: image(),
      alt: z.string().default(''),
      inferSize: z.boolean().default(false),
      width: z.number().optional(),
      height: z.number().optional(),
      color: z.string().default('#ffffff')
    }).optional(),
    tags: z.array(z.string()),
    language: z.string().optional(),
    draft: z.boolean().default(false),
    comment: z.boolean().default(true)
  })
})

const docs = defineCollection({
  loader: glob({
    base: './src/content/docs',
    pattern: '**/*.{md,mdx}'
  }),
  schema: z.object({
    title: z.string().max(60),
    description: z.string().max(160),
    publishDate: z.coerce.date().optional(),
    updatedDate: z.coerce.date().optional(),
    tags: z.array(z.string()),
    draft: z.boolean().default(false),
    order: z.number().default(999)
  })
})

export const collections = { blog, docs }
```

```typescript
export async function getBlogCollection(contentType: 'blog' | 'docs') {
  const allPosts = await getCollection(contentType)
  return import.meta.env.PROD
    ? allPosts.filter(post => !post.data.draft)
    : allPosts

// 按日期排序（降序）
export function sortMDByDate(collections: Array<any>) {
  return collections.sort((a, b) => {
    const aDate = new Date(a.data.publishDate).getTime()
    const bDate = new Date(b.data.publishDate).getTime()
    return bDate - aDate
  })
}

// 按年份分组
export function groupCollectionsByYear(collections: Array<any>) {
  return collections.reduce((acc, post) => {
    const year = new Date(post.data.publishDate).getFullYear()
    if (!acc[year]) acc[year] = []
    acc[year].push(post)
    return acc
  }, {})
}

// 获取所有标签
export function getAllTags(collections: Array<any>) {
  return collections.flatMap(post => post.data.tags)
}

// 获取唯一标签
export function getUniqueTags(collections: Array<any>) {
  return [...new Set(getAllTags(collections))]
}

// 获取标签及计数
export function getUniqueTagsWithCount(collections: Array<any>) {
  const allTags = getAllTags(collections)
  const tagCounts = allTags.reduce((acc, tag) => {
    acc[tag] = (acc[tag] || 0) + 1
    return acc
  }, {})

  return Object.entries(tagCounts)
    .map(([tag, count]) => ({ tag, count }))
    .sort((a, b) => b.count - a.count)
}
```

### 6.10 RSS 生成 

**位置：** `src/pages/rss.xml.ts`

**实现细节：**

```typescript
import rss from '@astrojs/rss'
import { getBlogCollection, sortMDByDate } from 'astro-pure/utils'
import { getImage } from 'astro:assets'
import MarkdownIt from 'markdown-it'
import config from '@/site-config'

const parser = new MarkdownIt()

export async function GET(context) {
  const blog = await getBlogCollection('blog')
  const sortedPosts = sortMDByDate(blog)

  const items = await Promise.all(
    sortedPosts.map(async (post) => {
      // 处理 Markdown 内容
      const content = parser.render(post.body)

      // 处理封面图
      let heroImage = undefined
      if (post.data.heroImage?.src) {
        const imageData = await getImage({
          src: post.data.heroImage.src,
          width: 1200
        })
        heroImage = new URL(imageData.src, context.site).href
      }

      return {
        title: post.data.title,
        pubDate: post.data.publishDate,
        description: post.data.description,
        link: `/blog/${post.id}/`,
        content,
        enclosure: heroImage ? {
          url: heroImage,
          type: 'image/jpeg'
        } : undefined
      }
    })
  )

  return rss({
    title: config.theme.title,
    description: config.theme.description,
    site: context.site,
    items,
    customData: `<language>${config.theme.locale.lang}</language>`,
    stylesheet: '/pretty-feed-v3.xsl'
  })
}
```

**特性：**
- Markdown 自动转 HTML
- 封面图作为 enclosure（支持 RSS 阅读器显示）
- 图片路径自动解析（相对路径 → 绝对 URL）
- Pretty Feed 样式表（浏览器友好显示）
